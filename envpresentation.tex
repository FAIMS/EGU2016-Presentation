function userdata.makeqr(txt,ht)
  foo = os.tmpname()..".png"
  os.execute("qrencode "..txt.." -s 20 -d 600 -o " .. foo)

  context.externalfigure(
    {foo},{height=ht.."mm"})
end

\def\maketitle{%
  %\startalignment[left]
    \blank[force,2*big]
      {\tfd \@title}
    \blank[2*medium]
      {\tfa \date[\@docdate][day,~,month,~,year]}
    \blank
      {\tfx \@docver}
    \blank[3*medium]
  %\stopalignment
}
